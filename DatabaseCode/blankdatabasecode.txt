USE wwdb

CREATE TABLE user
	(username	varchar(45),
	password	varchar(45),
	email		varchar(45),
	primary key(username));

CREATE TABLE usergroup
	(gid		int,
	gname		varchar(45),
	primary key(gid));

CREATE TABLE movie
	(mid		int,
	title		varchar(45),
	year		varchar(45),
	genre		varchar(45),
	rating		int,
	primary key(mid));

CREATE TABLE member
	(username	varchar(45),
	gid		int,
	primary key(username, gid));


CREATE TABLE admin
	(username	varchar(45),
	gid		int,
	primary key(username, gid));

CREATE TABLE rates
	(username	varchar(45),
	mid		int,
	yn		int,
	primary key(username, mid));

CREATE TABLE uwatched
	(username	varchar(45),
	mid		int,
	primary key(username, mid));

CREATE TABLE gwatched
	(gid		int,
	mid		int,
	primary key(gid, mid));
